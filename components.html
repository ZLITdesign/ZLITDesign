layui.use('laypage', function(){
  var laypage = layui.laypage;
  //完整功能
  laypage.render({
    elem: 'paging-1',
    count: 100,
    theme: '#108EE9',
    prev: '<',
    next: '>',
    layout: ['prev', 'page', 'next', 'count', 'limit', 'skip'],
    jump: function(obj){
      console.log(obj)
    }
  });
});

layui.use('laypage', function(){
  var laypage = layui.laypage;
  //简易样式一
  laypage.render({
    elem: 'paging-2',

    theme: '#108EE9',
    groups: 10,
    prev: '<',
    next: '>',
  });

  //简易样式二
  laypage.render({
    elem: 'paging-3',
    count: 70, //数据总数
    theme: '#fff',
    groups: 10,

操作确认框

            
操作需要用户进一步的确认时，询问或提示用户。

            
1.模态对话框：位于界面中心，可通过确认或取消按钮进行关闭，关闭前不能进行其他操作。

            
2.模态对话框的确认按钮文字可根据具体操作改变，如“提交”“删除”让用户明确当前操作。

            
3.气泡浮层：在目标元素附近弹出浮层提示，选择操作后消失或在空白处点击消失，交互形式更轻量；可用在列表页中删除单行数据时。

layui.use('layer', function(){
  var layer = layui.layer;
  //提交
  var index = layer.open({
    type: 0,
    icon: 4,
    title:'普通操作标题',
    offset: 'auto',
    content: '继续此操作的一些提示文字信息',
    btn: ['提交','取消'],
    yes: function(){
      //点击提交后。。。
    }
  });
  layer.style(index,{
    borderRadius:'6px',
    overflow:'hidden',
  });

  //删除
  var index = layer.open({
    type: 0,
    icon: 0,
    title:'普通操作标题',
    offset: 'auto',
    content: '继续此操作的一些提示文字信息',
    btn: ['删除','取消'],
    yes: function(){
      //点击删除后。。。
    }
  });
  layer.style(index,{
    borderRadius:'6px',
    overflow:'hidden',
  });

  //操作成功
  var index = layer.open({
    type: 0,
    icon: 1,
    title:'普通操作标题',
    offset: 'auto',
    time: 5000,
    content: '继续此操作的一些提示文字信息，5秒后自动关闭',
    btn: ['关闭'],
    yes: function(){
      //点击关闭后。。。
    }
  });
  layer.style(index,{
    borderRadius:'6px',
    overflow:'hidden',
  });

  //操作失败
  var index = layer.open({
    type: 0,
    icon: 2,
    title:'普通操作标题',
    offset: 'auto',
    content: '继续此操作的一些提示文字信息',
    btn: ['关闭'],
    yes: function(){
      layer.closeAll();
    }
  });
  layer.style(index,{
    borderRadius:'6px',
    overflow:'hidden',
  });

1.数字需定义最大值。

            
2.可实现数字以外的文本内容，如：new，hot等状态标记。

            
3.可简单直接显示小红点无误任何文字，提示状态有更新。

            
4.也可单独使用，用于某菜单下的数量标记。
